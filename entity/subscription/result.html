Result

Once you've got the result you should always check if an error has occurred, if not, retrieve the resource.

use Railken\Amethyst\Managers\SubscriptionManager;

$manager = new SubscriptionManager();

$result = $manager->create([
    "consume_rule" => [
        "name" => "Jeremy Pollich",
        "description" => "Explicabo rerum quod minus. Maiores inventore dolores hic error aut perspiciatis et id. Distinctio eum nesciunt est qui dolorem est. Sit dolorum delectus ut.",
        "class_name" => "Railken\\Amethyst\\ConsumeRules\\BaseConsumeRule",
        "payload" => [
            "x" => 1
        ]
    ],
    "status" => "pending",
    "metadata" => "day:monday",
    "subscriptionable_type" => "foo",
    "subscriptionable" => [
        "name" => "Korbin Brakus",
        "description" => "Repellat amet molestias error sint. Officiis ea ipsum aut sed. Quam vitae et voluptatibus ipsum. Quia voluptas dolorem voluptatibus voluptatem.",
        "bar" => [
            "name" => "Cade Murazik",
            "description" => "Placeat saepe omnis est iure. Iusto quis sint nemo aliquam expedita autem praesentium. Consequuntur ea nam non. Doloremque cupiditate itaque ipsum libero vel maxime iste."
        ]
    ]
]);

if ($result->ok()) {

    $resource = $result->getResource();

} else {

    // Loop through all errors
    $result->getErrors()->map(function($error) {
        return $error->toArray();
    }))

    // Retrieve an array of all errors
    $result->getSimpleErrors();

    /* The result is something like this:

        [0] => Array
            (
                [code] => FIELD_NOT_DEFINED
                [label] => field
                [message] => The field is required
                [value] =>
            )
    */

}
